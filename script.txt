<script>
  function showSong() {
    const mood = document.getElementById("moodSelector").value;
    const preview = document.getElementById("songPreview");

    let embed = "";

    if (mood === "sad") {
      embed = `<iframe width="100%" height="315" src="https://www.youtube.com/embed/Umqb9KENgmk" title="Uska Hi Bana" frameborder="0" allowfullscreen></iframe>`;
    } else if (mood === "romantic") {
      embed = `<iframe width="100%" height="315" src="https://www.youtube.com/embed/Umqb9KENgmk" title="Tum Hi Ho" frameborder="0" allowfullscreen></iframe>`;
    } else if (mood === "hopeful") {
      embed = `<iframe width="100%" height="315" src="https://www.youtube.com/embed/0G3_kG5FFfQ" title="Zinda" frameborder="0" allowfullscreen></iframe>`;
    } else if (mood === "nostalgic") {
      embed = `<iframe width="100%" height="315" src="https://www.youtube.com/embed/7ZzvGZ8Z4xI" title="Purono Sei Diner Kotha" frameborder="0" allowfullscreen></iframe>`;
    }

    preview.innerHTML = embed;
  }
</script>
